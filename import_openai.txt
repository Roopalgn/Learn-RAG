# Import the OpenAI class from the openai Python library
# This library allows Python to communicate with OpenAI's servers (LLM API)

from openai import OpenAI

# Create a client object that will handle communication with the API
# The API key authenticates you and allows OpenAI to know who is making the request

client = OpenAI(api_key="your api key")

# Send a request to the LLM
# The result returned by the API will be stored inside the variable called "response"

response = client.chat.completions.create(

    # Choose which language model to use
    # gpt-4o-mini is a fast and cheap model suitable for learning and simple applications

   model="gpt-4o-mini",

    # Messages represent the conversation sent to the model
    # The API expects a list (array) of message objects

   messages=[

        # Each message has two important fields:
        # role → who is speaking
        # content → the text of the message
        #"user" means this message is coming from the user (the prompt)

       {"role": "user", "content": "Explain what AI is in one sentence"}
   ]
)

# The API returns a large structured response (JSON-like object)
# The generated text from the AI is located here:
#
# response
#   └── choices
#         └── message
#               └── content
#
# choices[0] means we take the first generated answer
# This program SIMULATES the JSON response that the OpenAI API would normally return.
# We do this so we can learn how to work with LLM responses even without API credits.

print(response.choices[0].message.content)

# This prints only the AI's text response to the terminal
# Example output:
# Artificial intelligence is the ability of machines to perform tasks that normally require human intelligence.


# Here we create a Python dictionary that looks like the JSON response from the API
# JSON from APIs is automatically converted into Python dictionaries

response = {
    "id": "chatcmpl-abc123",
    "object": "chat.completion",
    "model": "gpt-4o-mini",

    # "choices" is a list because the API can generate multiple responses
    "choices": [
        {
            "index": 0,

            # This is the actual generated message
            "message": {
                "role": "assistant",

                # The text generated by the LLM
                "content": "Artificial Intelligence is the ability of machines to perform tasks that normally require human intelligence."
            },

            "finish_reason": "stop"
        }
    ],

    # Token usage information returned by the API
    "usage": {
        "prompt_tokens": 12,
        "completion_tokens": 18,
        "total_tokens": 30
    }
}


# Print the entire response so we can see the structure
print("FULL RESPONSE:\n")
print(response)


print("\n---------------------\n")


# Extract the generated text from the response
# Step-by-step path:
# response -> choices -> first element -> message -> content

generated_text = response["choices"][0]["message"]["content"]


# Print only the AI's generated text
print("AI RESPONSE:\n")
print(generated_text)


types of role

system
Gives instructions that control how the AI should behave.
Sets rules like tone, style, expertise, or restrictions.
Applies to the whole conversation.

user
Represents the human’s prompt or question.
Contains what the user is asking the AI.
The model mainly responds to the latest user message.

assistant
Represents responses previously generated by the AI.
Used to maintain conversation context.
Helps the model continue the discussion correctly.


Safe Response Parsing
When you get a response from an API, sometimes things can go wrong:
The API might fail
The response may not contain "choices"
The list might be empty

If your code directly does:
response["choices"][0]["message"]["content"]
the program can crash with an error.
So real programs check the response first.

# Simulated API response (like what an LLM would return)
response = {
    "choices": [
        {
            "message": {
                "role": "assistant",
                "content": "Artificial Intelligence is the simulation of human intelligence in machines."
            }
        }
    ]
}


# Check if the response contains the key "choices"
# This prevents errors if the API response is missing expected fields
if "choices" in response:

    # Check if the list actually contains at least one result
    if len(response["choices"]) > 0:

        # Extract the AI's generated text
        ai_text = response["choices"][0]["message"]["content"]

        # Print the response
        print("AI Response:", ai_text)

    else:
        print("No responses returned from the API.")

else:
    print("Invalid API response format.")
